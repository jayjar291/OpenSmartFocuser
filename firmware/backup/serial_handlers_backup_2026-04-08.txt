void handleGetPosition(HardwareSerial& serial, const char* parameters, size_t parametersLength) {
	(void)parameters;
	if (hasParameters(parametersLength)) {
		serial.println(":ERR#");
		return;
	}
	char response[24] = {0};
	snprintf(response, sizeof(response), "%ld#", static_cast<long>(Movement::getCurrentPositionSteps()));
	serial.println(response);
}

void handleHome(HardwareSerial& serial, const char* parameters, size_t parametersLength) {
	(void)parameters;
	if (hasParameters(parametersLength)) {
		serial.println(":ERR#");
		return;
	}
	Movement::startHoming();
	serial.println(":ACK#");
	gHomeCompletionPending = Movement::isBusy();

	if (!gHomeCompletionPending) {
		serial.println(":HD#");
	}
}

void handleGetTarget(HardwareSerial& serial, const char* parameters, size_t parametersLength) {
	(void)parameters;
	if (hasParameters(parametersLength)) {
		serial.println(":ERR#");
		return;
	}
	char response[56] = {0};
	snprintf(response, sizeof(response), ":TI%.6f,%.6f#", idleMapCenterRaDeg, idleMapCenterDecDeg);
	serial.println(response);
}

void handleSetTarget(HardwareSerial& serial, const char* parameters, size_t parametersLength) {
	float parsedRa = 0.0f;
	float parsedDec = 0.0f;
	char parsedName[kMaxTargetNameLength + 1] = {0};

	if (!parseSetTarget(parameters, parametersLength, parsedRa, parsedDec, parsedName, sizeof(parsedName))) {
		serial.println(":ERR#");
		return;
	}

	idleMapCenterRaDeg = parsedRa;
	idleMapCenterDecDeg = parsedDec;
	std::strncpy(gTargetName, parsedName, sizeof(gTargetName) - 1);
	gTargetName[sizeof(gTargetName) - 1] = '\0';

	serial.println(":ACK#");
}

void handleGotoPreset(HardwareSerial& serial, const char* parameters, size_t parametersLength) {
	uint8_t id = 0;
	if (!parsePresetId(parameters, parametersLength, id)) {
		serial.println(":ERR#");
		return;
	}

	const long targetSteps = loadPresetPositionById(id);
	const int32_t currentSteps = Movement::getCurrentPositionSteps();
	const bool wasEnabled = Movement::isMotorEnabled();

	DebugSerial::printFramedValue("PG id=", static_cast<int>(id), "");
	DebugSerial::printFramedValue("PG target=", targetSteps, "");
	DebugSerial::printFramedValue("PG current=", currentSteps, "");
	DebugSerial::printFramedValue("PG motorEnabled=", wasEnabled ? 1 : 0, "");

	if (!wasEnabled) {
		Movement::toggleMotorEnabled();
	}

	Movement::moveToPosition(static_cast<int32_t>(targetSteps));
	serial.println(":ACK#");
}

void handleGetPreset(HardwareSerial& serial, const char* parameters, size_t parametersLength) {
	uint8_t id = 0;
	if (!parsePresetId(parameters, parametersLength, id)) {
		serial.println(":ERR#");
		return;
	}
	const long pos = loadPresetPositionById(id);
	char response[32] = {0};
	snprintf(response, sizeof(response), ":PR%u,%ld#", static_cast<unsigned>(id), pos);
	serial.println(response);
}

float wrapRaDegrees(float raDeg) {
	while (raDeg >= 360.0f) {
		raDeg -= 360.0f;
	}
	while (raDeg < 0.0f) {
		raDeg += 360.0f;
	}
	return raDeg;
}

float clampDecDegrees(float decDeg) {
	if (decDeg > 90.0f) {
		return 90.0f;
	}
	if (decDeg < -90.0f) {
		return -90.0f;
	}
	return decDeg;
}

bool hasParameters(size_t parametersLength) {
	return parametersLength > 0;
}

bool parseSetTarget(const char* parameters,
										size_t parametersLength,
										float& raOut,
										float& decOut,
										char* nameOut,
										size_t nameOutSize) {
	if (parameters == nullptr || parametersLength == 0 || nameOut == nullptr || nameOutSize == 0) {
		return false;
	}

	char payload[kMaxCommandLength + 1] = {0};
	if (parametersLength > kMaxCommandLength) {
		return false;
	}
	std::memcpy(payload, parameters, parametersLength);
	payload[parametersLength] = '\0';

	char* firstComma = std::strchr(payload, ',');
	if (firstComma == nullptr) {
		return false;
	}
	*firstComma = '\0';

	char* secondComma = std::strchr(firstComma + 1, ',');

	char* raEnd = nullptr;
	const float parsedRa = std::strtof(payload, &raEnd);
	if (raEnd == payload || *raEnd != '\0') {
		return false;
	}

	char* decStart = firstComma + 1;
	if (*decStart == '\0') {
		return false;
	}

	if (secondComma != nullptr) {
		*secondComma = '\0';
	}

	char* decEnd = nullptr;
	const float parsedDec = std::strtof(decStart, &decEnd);
	if (decEnd == decStart || *decEnd != '\0') {
		return false;
	}

	nameOut[0] = '\0';
	if (secondComma != nullptr) {
		const char* nameStart = secondComma + 1;
		const size_t nameLength = std::strlen(nameStart);
		if (nameLength > kMaxTargetNameLength || nameLength >= nameOutSize) {
			return false;
		}
		std::strncpy(nameOut, nameStart, nameOutSize - 1);
		nameOut[nameOutSize - 1] = '\0';
	}

	raOut = wrapRaDegrees(parsedRa);
	decOut = clampDecDegrees(parsedDec);
	return true;
}

bool parsePresetId(const char* parameters, size_t parametersLength, uint8_t& idOut) {
	if (parameters == nullptr || parametersLength == 0 || parametersLength > 3) {
		return false;
	}

	char payload[4] = {0};
	std::memcpy(payload, parameters, parametersLength);

	char* idEnd = nullptr;
	const unsigned long parsed = std::strtoul(payload, &idEnd, 10);
	if (idEnd == payload || *idEnd != '\0' || parsed == 0 || parsed > 255) {
		return false;
	}
	idOut = static_cast<uint8_t>(parsed);
	return true;
}

void resetParser() {
	gCommandLength = 0;
	gCapturing = false;
}

void dispatchCommand(const char* frame, size_t frameLength) {
	if (!SerialCommandIndex::dispatch(*gSerial, frame, frameLength)) {
		gSerial->println(":ERR#");
	}
}
